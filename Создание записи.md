## Пример создания подразделения

```js
_doc_org = OpenNewDoc( 'x-local://wtv/wtv_org.xmd' );
_doc_org.BindToDb(DefaultDb);
_doc_org.TopElem.disp_name = 'Рога и Копыта';
_doc_org.TopElem.name = 'Рога и Копыта';
_doc_org.Save();

counter = 10;
for(i = 1; i <= counter; i++)
{
    _doc = OpenNewDoc( 'x-local://wtv/wtv_subdivision.xmd' );
    _doc.BindToDb(DefaultDb);
    _doc.TopElem.org_id = _doc_org.DocID;

    //_doc.TopElem.name = 'Подразделение.' + Random( 1, 4000000000 );
    _doc.TopElem.name = 'Подразделение == ' + i;
    _doc.TopElem.status = 'active';
    _doc.TopElem.is_disbanded = false;

    _doc.TopElem.show_detailed = true;

    _doc.TopElem.show_children = false;

    _doc.TopElem.is_faculty = false;

  

    _doc.Save();

}
```
