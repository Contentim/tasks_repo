частый случай с курсами и тестами - в данном баге было исправление при ситуации, когда удалили запись сотрудника и отчет отваливается из-за того что не может выполниться запрос к каталогу collaborator - oTempLearning.person_id.ForeignElem.

```javascript
if(oTempLearning.person_id.HasValue)
{
	sQuery = 'for $elem_qc in collaborators where MatchSome( $elem_qc/id, ( ' + oTempLearning.person_id + ' ) ) return $elem_qc';
	arrPerson = ArrayOptFirstElem(tools.xquery( sQuery ));
	if(arrPerson != undefined)
	{
		oTempLearning.person_fullname = oTempLearning.person_id.ForeignElem.fullname;
		oTempLearning.person_code = oTempLearning.person_id.ForeignElem.code;
		oTempLearning.person_position_name = oTempLearning.person_id.ForeignElem.position_name;
		oTempLearning.person_org_name = oTempLearning.person_id.ForeignElem.org_name;
		oTempLearning.person_subdivision_name = oTempLearning.person_id.ForeignElem.position_parent_name;

	} else {
		// например запись сотрудника удалили из таблицы collaborator
		object = tools.open_doc(oTempLearning.id.Value);
		if ( object != undefined )
		{
			oTempLearning.person_fullname = object.TopElem.person_fullname;
			oTempLearning.person_code = object.TopElem.person_code;
			oTempLearning.person_position_name = object.TopElem.person_position_name;
			oTempLearning.person_org_name = object.TopElem.person_org_name;
			oTempLearning.person_subdivision_name = object.TopElem.person_subdivision_name;
		}
	}
} else {
oTempLearning.person_fullname = oTempLearning.person_code = oTempLearning.person_position_name = oTempLearning.person_org_name = oTempLearning.person_subdivision_name = '';
}
```
